45  /**
46   * Default byte buffer backed by a list of arrays to overcome the size
47   * limitations of a usual Java byte buffer. It can theoretically allocate
48   * MAX_INT*MAX_INT bytes, because it uses long for size related functions
49   * the maximum capacity is MAX_LONG. The capacity is allocated in
50   * chunks as soon as needed and not at once to prevent unnecessary
51   * memory allocation.
52   *
53   *
54   */
55  public class DefaultByteBuffer implements ByteBuffer {
56  
57      private final List<byte[]> buffer;
58  
59      private final int chunkSize;
60      private final long capacity;
61      private final int lastRowSize;
62  
63      private int readColPointer;
64      private int readRowPointer;
65      private int writeColPointer;
66      private int writeRowPointer;
67  
68      /**
69       * Creates the buffer with the given capacity and
70       * a default chunk size of MAX_INT.
71       * @param capacity the buffer capacity
72       */
73      public DefaultByteBuffer(final long capacity) {
74          this(capacity, Integer.MAX_VALUE);
75      }

77      /**
78       * Creates the buffer with the given capacity and the
79       * given chunk size used to allocate memory.
80       *
81       * @param capacity the buffer capacity
82       * @param chunkSize the memory allocation chunk size
83       */
84      public DefaultByteBuffer(final long capacity, final int chunkSize) {
85          if (chunkSize < 0 || capacity < 0) {
86              this.chunkSize = 0;
87              this.capacity = 0;
88          } else {
89              if (capacity <= chunkSize) {
90                  this.capacity = capacity;
91                  this.chunkSize = (int) capacity;
92              } else {
93                  this.capacity = capacity;
94                  this.chunkSize = chunkSize;
95              }
96          }
97  
98          if (this.capacity <= this.chunkSize) {
99              this.buffer = new ArrayList<>(1);
100             this.buffer.add(new byte[(int) this.capacity]);
101             this.lastRowSize = (int) this.capacity;
102 
103         } else {
104 
105             // count last row size
106             this.lastRowSize = (int) (capacity % this.chunkSize);
107 
108             // count columns
109             long cols = (capacity / this.chunkSize);
110             cols = (this.lastRowSize > 0) ? cols + 1 : cols;
111 
112             if (cols <= Integer.MAX_VALUE) {
113 
114                 this.buffer = new ArrayList<>((int) cols);
115                 this.buffer.add(new byte[this.chunkSize]);
116 
117             } else {
118 
119                 throw new IllegalArgumentException(
120                         "Capacity is to large and cannot "
121                         + "be supported by this implementation.");
122 
123             }
124         }
125 
126         this.readColPointer = 0;
127         this.readRowPointer = -1;
128         this.writeColPointer = 0;
129         this.writeRowPointer = -1;
130     }

132     @Override
133     public void write(final byte[] array) {
134         if (array == null) {
135             throw new NullPointerException("Array cannot be null.");
136         }
137 
138         int length = array.length;
139 
140         this.checkWriteBufferOverflow(length);
141 
142         int index = 0;
143 
144         if (this.writeRowPointer >= (this.chunkSize - length)) {
145             while (length > 0) {
146                 this.writeRowPointer++;
147 
148                 if (this.writeRowPointer >= this.chunkSize) {
149                     if ((this.writeColPointer + 1)
150                             < (capacity / this.chunkSize)) {
151                         this.buffer.add(new byte[this.chunkSize]);
152                     } else {
153                         this.buffer.add(new byte[this.lastRowSize]);
154                     }
155 
156                     this.writeRowPointer = 0;
157                     this.writeColPointer++;
158                 }
159 
160                 int cpLength = ((this.chunkSize - this.writeRowPointer)
161                         < length) ? this.chunkSize
162                         - this.writeRowPointer
163                         : length;
164 
165                 System.arraycopy(array, index,
166                         this.buffer.get(this.writeColPointer),
167                         this.writeRowPointer, cpLength);
168                 length -= cpLength;
169                 index += cpLength;
170                 this.writeRowPointer += cpLength - 1;
171             }
172         } else {
173             if (length > 0) {
174                 this.writeRowPointer++;
175                 System.arraycopy(array, index,
176                         this.buffer.get(this.writeColPointer),
177                         this.writeRowPointer, length);
178                 this.writeRowPointer += (length - 1);
179             }
180         }
181     }

183     @Override
184     public void write(final ByteBuffer byteBuffer) {
185 
186         if (byteBuffer == null) {
187             throw new NullPointerException("Buffer cannot be null.");
188         }
189 
190         this.checkWriteBufferOverflow(byteBuffer.bytesWritten());
191 
192         while (byteBuffer.bytesWritten() > byteBuffer.bytesRead()) {
193             if ((byteBuffer.bytesWritten() - byteBuffer.bytesRead())
194                     > this.chunkSize) {
195                 this.write(byteBuffer.read(this.chunkSize));
196             } else {
197                 this.write(byteBuffer.read(
198                         (int) (byteBuffer.bytesWritten()
199                                 - byteBuffer.bytesRead())));
200             }
201         }
202 
203     }

205     @Override
206     public void writeByte(final byte b) {
207         // check bounds
208         this.checkWriteBufferOverflow(1);
209 
210         this.writeRowPointer++;
211 
212         if (this.writeRowPointer >= this.chunkSize) {
213             if ((this.writeColPointer + 1) < (capacity / this.chunkSize)) {
214                 this.buffer.add(new byte[this.chunkSize]);
215             } else {
216                 this.buffer.add(new byte[this.lastRowSize]);
217             }
218 
219             this.writeRowPointer = 0;
220             this.writeColPointer++;
221         }
222 
223         this.buffer.get(this.writeColPointer)[this.writeRowPointer] = (byte) b;
224     }

274     @Override
275     public void writeDigits(final long number, final byte length) {
276         if (length <= 0) {
277             return;
278         }
279 
280         if (length > 8) {
281             throw new IllegalArgumentException("Supplied length " + length
282                     + " is to long.");
283         }
284 
285         byte[] b = new byte[length];
286 
287         byte l = (byte) (length - 1);
288 
289         for (int i = 0; i < length; i++) {
290 
291             b[i] = (byte) (number >>> (8 * (l - i)));
292 
293         }
294         this.write(b);
295     }

297     @Override
298     public byte[] read(final int length) {
299         if (length <= 0) {
300             // obviously we cannot read anything so retrun 0 length;
301             return new byte[0];
302         }
303 
304         int length2 = length;
305 
306         this.checkReadBufferUnderflow(length2);
307 
308         byte[] array = new byte[length];
309 
310         int index = 0;
311 
312         if (this.readRowPointer >= (this.chunkSize - length2)) {
313             while (length2 > 0) {
314                 this.readRowPointer++;
315                 if (this.readRowPointer >= this.chunkSize) {
316                     this.readRowPointer = 0;
317                     this.readColPointer++;
318                 }
319 
320                 int cpLength = ((this.chunkSize - this.readRowPointer)
321                         < length2) ? this.chunkSize
322                         - this.readRowPointer
323                         : length2;
324 
325                 System.arraycopy(this.buffer.get(this.readColPointer),
326                         this.readRowPointer, array, index, cpLength);
327                 length2 -= cpLength;
328                 index += cpLength;
329                 this.readRowPointer += cpLength - 1;
330             }
331         } else {
332             this.readRowPointer++;
333             System.arraycopy(this.buffer.get(this.readColPointer),
334                     this.readRowPointer, array, index, length2);
335             this.readRowPointer += (length2 - 1);
336         }
337 
338         return array;
339     }

341     @Override
342     public ByteBuffer read(final long length) {
343 
344         if (length <= 0) {
345             // obviously we cannot read anything so retrun 0 length;
346             return new DefaultByteBuffer(0);
347         }
348 
349         this.checkReadBufferUnderflow(length);
350 
351         ByteBuffer b = new DefaultByteBuffer(length);
352 
353         while (b.bytesWritten() < length) {
354             if (length - b.bytesWritten() > this.chunkSize) {
355                 b.write(this.read(this.chunkSize));
356             } else {
357                 b.write(this.read((int) (length - b.bytesWritten())));
358             }
359         }
360 
361         return b;
362 
363     }

365     @Override
366     public byte readByte() {
367         // check bounds
368         this.checkReadBufferUnderflow(1);
369 
370         this.readRowPointer++;
371 
372         if (this.readRowPointer >= this.chunkSize) {
373             this.readRowPointer = 0;
374             this.readColPointer++;
375         }
376 
377         return this.buffer.get(this.readColPointer)[this.readRowPointer];
378     }

454     @Override
455     public long revertWrite(final long length) {
456         if (length < 0) {
457             throw new IllegalArgumentException(
458                     "Parameter must be a positiv integer.");
459         }
460 
461         long currentFullPos = this.bytesWritten();
462 
463         long revertPos = ((currentFullPos - length) < 0)
464                 ? 0 : currentFullPos - length;
465 
466         // count last row size
467         int revertedLastRowSize = (int) (revertPos % this.chunkSize);
468 
469         // count columns
470         int revertedCols = (int) (revertPos / this.chunkSize);
471 
472         if (this.writeColPointer > revertedCols) {
473             for (int i = this.writeColPointer; i > revertedCols; --i) {
474                 this.buffer.remove(i);
475             }
476             this.writeRowPointer = this.chunkSize;
477         }
478 
479         this.writeColPointer = revertedCols;
480 
481         if (this.writeRowPointer > revertedLastRowSize) {
482             byte[] array = new byte[this.chunkSize];
483             System.arraycopy(this.buffer.get(this.writeColPointer), 0,
484                     array, 0, revertedLastRowSize);
485             this.buffer.set(this.writeColPointer, array);
486         }
487         this.writeRowPointer = revertedLastRowSize - 1;
488 
489         return this.bytesWritten();
490     }

492     @Override
493     public long revertRead(final long length) {
494         if (length < 0) {
495             throw new IllegalArgumentException(
496                     "Parameter must be a positiv integer.");
497         }
498 
499         long currentFullPos = (this.readColPointer * this.chunkSize)
500                 + this.readRowPointer + 1;
501 
502         long revertPos = ((currentFullPos - length) < 0)
503                 ? 0 : currentFullPos - length;
504 
505         // count last row size
506         int revertedLastRowSize = (int) (revertPos % this.chunkSize) - 1;
507 
508         // count columns
509         int revertedCols = (int) (revertPos / this.chunkSize);
510 
511         this.readColPointer = revertedCols;
512         this.readRowPointer = revertedLastRowSize;
513 
514         return this.bytesRead();
515     }

517     @Override
518     public void clear() {
519 
520         this.readColPointer = 0;
521         this.readRowPointer = -1;
522         this.writeColPointer = 0;
523         this.writeRowPointer = -1;
524 
525         this.buffer.clear();
526 
527         if (capacity <= this.chunkSize) {
528             this.buffer.add(new byte[(int) capacity]);
529         } else {
530 
531             // count columns
532             long cols = (capacity / this.chunkSize);
533             cols = (this.lastRowSize > 0) ? cols + 1 : cols;
534 
535             if (cols <= Integer.MAX_VALUE) {
536 
537                 this.buffer.add(new byte[this.chunkSize]);
538 
539             } else {
540 
541                 throw new IllegalArgumentException(
542                         "Capacity is to large and cannot be "
543                         + "supported by this implementation.");
544 
545             }
546         }
547 
548     }

589     /**
590      * Checks if the length of bytes can be written to
591      * the buffer without exceeding the available capacity.
592      *
593      * @param lookAhead the length of bytes to look ahead
594      */
595     private void checkWriteBufferOverflow(final long lookAhead) {
596         if (lookAhead < 0) {
597             throw new IllegalArgumentException(
598                     "Parameter must be a positiv integer.");
599         }
600 
601         long currentFullPos = (this.writeColPointer * this.chunkSize)
602                 + this.writeRowPointer + 1;
603 
604         if (currentFullPos > (this.capacity - lookAhead)) {
605             throw new BufferOverflowException();
606         }
607     }
608 
609     /**
610      * Checks if the length of bytes can be read from the buffer
611      * without exceeding the bytes written.
612      *
613      * @param lookAhead the length of bytes to look ahead
614      */
615     private void checkReadBufferUnderflow(final long lookAhead) {
616 
617         if (lookAhead < 0) {
618             throw new IllegalArgumentException(
619                     "Parameter must be a positiv integer.");
620         }
621 
622         long currentFullPosWrite = (this.writeColPointer * this.chunkSize)
623                 + this.writeRowPointer + 1;
624         long currentFullPosRead = (this.readColPointer * this.chunkSize)
625                 + this.readRowPointer + 1;
626 
627         if (currentFullPosRead > (currentFullPosWrite - lookAhead)) {
628             throw new BufferUnderflowException();
629         }
630     }

632     @Override
633     public int hashCode() {
634         final int prime = 31;
635         int result = 1;
636         result = prime * result
637                 + ((this.buffer == null) ? 0 : this.buffer.hashCode());
638         result = prime * result
639                 + (int) (this.capacity ^ (this.capacity >>> 32));
640         result = prime * result + this.chunkSize;
641         result = prime * result + this.lastRowSize;
642         result = prime * result + this.readColPointer;
643         result = prime * result + this.readRowPointer;
644         result = prime * result + this.writeColPointer;
645         result = prime * result + this.writeRowPointer;
646         return result;
647     }

649     @Override
650     public boolean equals(final Object obj) {
651         if (this == obj) {
652             return true;
653         }
654         if (obj == null) {
655             return false;
656         }
657         if (getClass() != obj.getClass()) {
658             return false;
659         }
660         DefaultByteBuffer other = (DefaultByteBuffer) obj;
661         if (this.buffer == null) {
662             if (other.buffer != null) {
663                 return false;
664             }
665         } else {
666             for (int i = 0; i < this.buffer.size(); i++) {
667                 if (this.buffer.get(i) == null) {
668                     if (other.buffer.get(i) != null) {
669                         return false;
670                     }
671                 } else {
672                     if (!(Arrays
673                             .equals(this.buffer.get(i), other.buffer.get(i)))) {
674                         return false;
675                     }
676                 }
677             }
678         }
679         if (this.capacity != other.capacity) {
680             return false;
681         }
682         if (this.chunkSize != other.chunkSize) {
683             return false;
684         }
685         if (this.lastRowSize != other.lastRowSize) {
686             return false;
687         }
688         if (this.readColPointer != other.readColPointer) {
689             return false;
690         }
691         if (this.readRowPointer != other.readRowPointer) {
692             return false;
693         }
694         if (this.writeColPointer != other.writeColPointer) {
695             return false;
696         }
697         if (this.writeRowPointer != other.writeRowPointer) {
698             return false;
699         }
700         return true;
701     }
