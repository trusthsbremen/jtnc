70  /**
71   * Default IMC handler to handle messages destined to/from an IMC.
72   *
73   *
74   */
75  public class DefaultImcHandler implements ImcHandler {
76  
77      private static final Logger LOGGER = LoggerFactory
78              .getLogger(DefaultImcHandler.class);
79  
80      private Map<Long, ImcAdapter> imAdapters;
81      private Map<Long, ImcConnectionAdapter> connections;
82  
83      private TncConnectionState state;
84      private boolean handshakeBegin;
85  
86      private final ImMessageRouter router;
87      private final ImAdapterManager<ImcAdapter> manager;
88  
89      private final ImcConnectionAdapterFactory connectionFactory;
90      private final ImcConnectionContext connectionContext;

92      /**
93       * Creates an IMC handler with the given IMC adapter manager
94       * to manage the IMC and the given connection factory to create IMC
95       * connections for every IMC with the given IMC connection
96       * context. The message router is used to route the message
97       * based on its component type address to IMC, that are interested
98       * in that type.
99       *
100      * @param manager the IMC manager
101      * @param connectionFactory the IMC connection factory
102      * @param connectionContext the IMC connection context
103      * @param router the message router
104      */
105     public DefaultImcHandler(final ImAdapterManager<ImcAdapter> manager,
106             final ImcConnectionAdapterFactory connectionFactory,
107             final ImcConnectionContext connectionContext,
108             final ImMessageRouter router) {
109         this.connectionFactory = connectionFactory;
110         this.manager = manager;
111         this.connectionContext = connectionContext;
112 
113         Map<Long, ImcAdapter> adapterList = this.manager.getAdapter();
114         Map<Long, ImcConnectionAdapter> connectionList = new HashMap<>(
115                 adapterList.size());
116 
117         for (Long key : adapterList.keySet()) {
118             connectionList.put(adapterList.get(key).getPrimaryId(),
119                     this.connectionFactory.createConnection(adapterList
120                             .get(key).getPrimaryId()));
121         }
122 
123         this.imAdapters = adapterList;
124         this.connections = connectionList;
125         this.router = router;
126 
127         this.state = DefaultTncConnectionStateEnum.HSB_CONNECTION_STATE_UNKNOWN;
128         this.handshakeBegin = false;
129 
130     }

132     @Override
133     public void setConnectionState(final TncConnectionState imConnectionState) {
134 
135         this.state = imConnectionState;
136 
137         if (imConnectionState.equals(DefaultTncConnectionStateEnum
138                 .TNC_CONNECTION_STATE_HANDSHAKE)) {
139 
140             this.handshakeBegin = true;
141             this.refreshAdapterEntries();
142             this.connectionContext.validate();
143         }
144 
145         for (Iterator<Entry<Long, ImcAdapter>> iter = this.imAdapters
146                 .entrySet().iterator(); iter.hasNext();) {
147             Entry<Long, ImcAdapter> entry = iter.next();
148 
149             try {
150 
151                 entry.getValue()
152                         .notifyConnectionChange(
153                                 this.connections.get(entry.getKey()),
154                                 imConnectionState);
155 
156             } catch (TerminatedException e) {
157                 this.connections.remove(entry.getKey());
158                 iter.remove();
159             } catch (TncException e) {
160                 if (e.getResultCode().equals(
161                         TncExceptionCodeEnum.TNC_RESULT_FATAL)) {
162                     this.connections.remove(entry.getKey());
163                     this.manager.notifyFatalError(entry.getKey(), e);
164                     iter.remove();
165 
166                 }
167                 LOGGER.error(e.getMessage(), e);
168             }
169 
170         }
171 
172         if (imConnectionState.id()
173                 == DefaultTncConnectionStateEnum
174                 .TNC_CONNECTION_STATE_DELETE.id()) {
175 
176             this.imAdapters.clear();
177             this.connections.clear();
178             this.handshakeBegin = false;
179             this.connectionContext.invalidate();
180         }
181 
182     }
183 
184     @Override
185     public List<TnccsMessage> requestMessages() {
186         this.checkState();
187 
188         for (Iterator<Entry<Long, ImcAdapter>> iter = this.imAdapters
189                 .entrySet().iterator(); iter.hasNext();) {
190             Entry<Long, ImcAdapter> entry = iter.next();
191 
192             try {
193 
194                 entry.getValue().beginHandshake(
195                         this.connections.get(entry.getKey()));
196 
197             } catch (TerminatedException e) {
198                 this.connections.remove(entry.getKey());
199                 iter.remove();
200             } catch (TncException e) {
201                 if (e.getResultCode().equals(
202                         TncExceptionCodeEnum.TNC_RESULT_FATAL)) {
203                     this.connections.remove(entry.getKey());
204                     this.manager.notifyFatalError(entry.getKey(), e);
205                     iter.remove();
206                 }
207                 LOGGER.error(e.getMessage(), e);
208             }
209         }
210 
211         if (this.handshakeBegin) {
212             this.handshakeBegin = false;
213         }
214 
215         return this.connectionContext.clearMessage();
216 
217     }

219     @Override
220     public List<TnccsMessage> forwardMessage(final TnccsMessage message) {
221         this.checkState();
222 
223         if (message == null || message.getValue() == null) {
224             LOGGER.debug("Because Message or message value is null,"
225                     + " it is ignored.");
226             return new ArrayList<TnccsMessage>();
227         }
228 
229         TnccsMessageValue value = message.getValue();
230 
231         if (value instanceof PbMessageValueIm) {
232 
233             PbMessageValueIm valueCast = (PbMessageValueIm) value;
234 
235             Set<Long> recipients = new HashSet<>();
236             if (valueCast.getImFlags().contains(PbMessageImFlagEnum.EXCL)) {
237                 try {
238                     Long recipient = this.router.findExclRecipientId(
239                         valueCast.getCollectorId(), valueCast.getSubVendorId(),
240                         valueCast.getSubType());
241                     recipients.add(recipient);
242                 } catch (NoRecipientFoundException e) {
243                     LOGGER.debug(e.getMessage());
244                 }
245             } else {
246                 try {
247                     recipients = this.router.findRecipientIds(
248                         valueCast.getSubVendorId(), valueCast.getSubType());
249                 } catch (NoRecipientFoundException e) {
250                     LOGGER.debug(e.getMessage());
251                 }
252             }
253 
254             for (Long recipientId : recipients) {
255                 if (this.imAdapters.containsKey(recipientId)) {
256 
257                     Boolean tncsFirstSupport = Boolean.FALSE;
258                     if (this.handshakeBegin) {
259 
260                         try {
261                             Object o = this.imAdapters
262                                     .get(recipientId)
263                                     .getAttribute(TncClientAttributeTypeEnum
264                                             .TNC_ATTRIBUTEID_IMC_SPTS_TNCS1);
265                             if (o instanceof Boolean) {
266                                 tncsFirstSupport = (Boolean) o;
267                             }
268                         } catch (TncException
269                                 | UnsupportedOperationException e) {
270 
271                             LOGGER.debug(
272                                     "TNCS first support was not identifiable. "
273                                     + "The feature is not used.",
274                                     e);
275                         }
276 
277                     }
278 
279                     try {
280 
281                         if (this.handshakeBegin && !tncsFirstSupport) {
282                             this.imAdapters.get(recipientId).beginHandshake(
283                                     this.connections.get(recipientId));
284                         } else {
285                             this.imAdapters.get(recipientId).handleMessage(
286                                     this.connections.get(recipientId), value);
287                         }
288 
289                     } catch (TerminatedException e) {
290                         this.imAdapters.remove(recipientId);
291                         this.connections.remove(recipientId);
292 
293                     } catch (TncException e) {
294                         if (e.getResultCode().equals(
295                                 TncExceptionCodeEnum.TNC_RESULT_FATAL)) {
296                             this.connections.remove(recipientId);
297                             this.manager.notifyFatalError(recipientId, e);
298                             this.imAdapters.remove(recipientId);
299 
300                         }
301                         LOGGER.error(e.getMessage(), e);
302                     }
303                 }
304             }
305 
306             if (this.handshakeBegin) {
307                 this.handshakeBegin = false;
308             }
309 
310         } else {
311             LOGGER.debug("Because Message is not of type "
312                     + PbMessageValueIm.class.getCanonicalName()
313                     + ", it is ignored.");
314         }
315 
316         return this.connectionContext.clearMessage();
317     }

319     @Override
320     public void dumpMessage(final TnccsMessage message) {
321         this.checkState();
322 
323         // dumping means, no messages should be send in response
324         if (this.connectionContext.isValid()) {
325             this.connectionContext.invalidate();
326         }
327 
328         if (message == null || message.getValue() == null) {
329             LOGGER.debug("Because Message or message value is null,"
330                     + " it is ignored.");
331             return;
332         }
333 
334         TnccsMessageValue value = message.getValue();
335 
336         if (value instanceof PbMessageValueIm) {
337 
338             PbMessageValueIm valueCast = (PbMessageValueIm) value;
339 
340             Set<Long> recipients = new HashSet<>();
341             if (valueCast.getImFlags().contains(PbMessageImFlagEnum.EXCL)) {
342                 try {
343                     Long recipient = this.router.findExclRecipientId(
344                             valueCast.getCollectorId(),
345                             valueCast.getSubVendorId(),
346                             valueCast.getSubType());
347                     recipients.add(recipient);
348                 } catch (NoRecipientFoundException e) {
349                     LOGGER.debug(e.getMessage());
350                 }
351 
352             } else {
353                 try {
354                     recipients = this.router.findRecipientIds(
355                             valueCast.getSubVendorId(), valueCast.getSubType());
356                 } catch (NoRecipientFoundException e) {
357                     LOGGER.debug(e.getMessage());
358                 }
359             }
360 
361             for (Long recipientId : recipients) {
362 
363                 if (this.imAdapters.containsKey(recipientId)) {
364                     try {
365 
366                         this.imAdapters.get(recipientId).handleMessage(
367                                 this.connections.get(recipientId), value);
368 
369                     } catch (TerminatedException e) {
370                         this.imAdapters.remove(recipientId);
371                         this.connections.remove(recipientId);
372 
373                     } catch (TncException e) {
374                         if (e.getResultCode().equals(
375                                 TncExceptionCodeEnum.TNC_RESULT_FATAL)) {
376                             this.connections.remove(recipientId);
377                             this.manager.notifyFatalError(recipientId, e);
378                             this.imAdapters.remove(recipientId);
379                         }
380                         LOGGER.error(e.getMessage(), e);
381                     }
382                 }
383             }
384         } else {
385             LOGGER.debug("Because Message is not of type "
386                     + PbMessageValueIm.class.getCanonicalName()
387                     + ", it is ignored.");
388         }
389 
390         // we do not want any messages. Just to be secure, discard them all
391         this.connectionContext.clearMessage();
392     }

394     @Override
395     public List<TnccsMessage> lastCall() {
396 
397         for (Iterator<Entry<Long, ImcAdapter>> iter = this.imAdapters
398                 .entrySet().iterator(); iter.hasNext();) {
399             Entry<Long, ImcAdapter> entry = iter.next();
400 
401             try {
402 
403                 entry.getValue().batchEnding(
404                         this.connections.get(entry.getKey()));
405 
406             } catch (TerminatedException e) {
407                 this.connections.remove(entry.getKey());
408                 iter.remove();
409             } catch (TncException e) {
410                 if (e.getResultCode().equals(
411                         TncExceptionCodeEnum.TNC_RESULT_FATAL)) {
412                     this.connections.remove(entry.getKey());
413                     this.manager.notifyFatalError(entry.getKey(), e);
414                     iter.remove();
415                 }
416                 LOGGER.error(e.getMessage(), e);
417             }
418         }
419 
420         return this.connectionContext.clearMessage();
421     }

423     /**
424      * Synchronizes the known list of IMC with the manager, which
425      * is responsible to manage all IMC.
426      */
427     private void refreshAdapterEntries() {
428         Map<Long, ImcAdapter> list = this.manager.getAdapter();
429 
430         Set<Long> oldKeys = new HashSet<>(this.imAdapters.keySet());
431         Set<Long> newKeys = new HashSet<>(list.keySet());
432         newKeys.removeAll(this.imAdapters.keySet());
433         oldKeys.removeAll(list.keySet());
434 
435         for (Long long1 : oldKeys) {
436             imAdapters.remove(long1);
437             connections.remove(long1);
438         }
439 
440         for (Long long1 : newKeys) {
441             ImcAdapter imAdapter = list.get(long1);
442             imAdapters.put(imAdapter.getPrimaryId(), imAdapter);
443             connections.put(imAdapter.getPrimaryId(), this.connectionFactory
444                     .createConnection(imAdapter.getPrimaryId()));
445         }
446     }
447 
448     /**
449      * Checks if the current state is a valid connection state.
450      */
451     private void checkState() {
452         if (this.state == null
453                 || this.state.equals(
454                         DefaultTncConnectionStateEnum
455                         .HSB_CONNECTION_STATE_UNKNOWN)) {
456             throw new IllegalStateException(
457                     "The handlers state cannot be:"
458                             + ((this.state != null) ? this.state
459                                     : DefaultTncConnectionStateEnum
460                                         .HSB_CONNECTION_STATE_UNKNOWN)
461                                         .toString());
462         }
463     }
